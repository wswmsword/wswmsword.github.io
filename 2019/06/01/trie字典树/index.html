  1. INSERT(s)
  2. QUERY(s)

INSERT(s)

u:=0
for i:=0 to (字符串s的长度-1)
   ch:=(s[i]的字母表编号)
   if c[u,ch]=0
      c[sz,...]全部置零
      c[u,ch]:=sz
      sz:=sz+1
   endif
   u:=c[u][ch]
   cnt[u]:=cnt[u]+1
endfor

QUERY(s)

for i:=0 to (字符串s的长度-1)
   ch:=(s[i]的字母表编号)
   if c[u,ch]=0||u><0&&cnt[u]<=1
      return i endif
   u=c[u,ch]
endfor
return n
